char data[100];

int apply(int a){
    return a^5;
}
void crypt(char* ptr, int len){
    int i;
    for(i = 0; i <len;i++){
        ptr[i] = apply(ptr[i]);
    }
}

int main(int argc, char** argv){
    memcpy(data,"M`iij%rjwia$",12);
    data[12] = 0;
    crypt(data, 12);
    printf("%s\n",data);
    crypt(data, 12);
}
